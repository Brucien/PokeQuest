        async function syncAllFromPokeAPI() {
            const syncBtn = document.getElementById('api-sync-btn');
            const progressWrapper = document.getElementById('api-progress-wrapper');
            const syncBanner = document.getElementById('sync-banner');
            
            syncBtn.disabled = true;
            progressWrapper.classList.remove('hidden');
            
            document.getElementById('api-progress-bar').style.width = '5%';
            document.getElementById('api-progress-percent').innerText = '5%';
            document.getElementById('api-progress-label').innerText = "Connecting to PokeAPI servers...";

            try {
                // Fetch the total list of species up to Gen 9 limit
                const response = await fetch('https://pokeapi.co/api/v2/pokemon-species/?limit=1025');
                if (!response.ok) throw new Error("Connection failed");
                const data = await response.json();
                
                document.getElementById('api-progress-bar').style.width = '20%';
                document.getElementById('api-progress-percent').innerText = '20%';
                document.getElementById('api-progress-label').innerText = "Sorting National Pokédex data...";

                const results = data.results;
                
                // Parse exact IDs from the URL and map
                const fetchedPokemon = results.map((item) => {
                    const urlParts = item.url.split('/').filter(Boolean);
                    const num = parseInt(urlParts[urlParts.length - 1]);
                    const nameRaw = item.name;
                    
                    // Clean names (remove hyphens)
                    const cleanName = nameRaw
                        .split('-')
                        .map(word => word.charAt(0).toUpperCase() + word.slice(1))
                        .join(' ');

                    // Define generation thresholds
                    let gen = 1;
                    if (num > 151) gen = 2;
                    if (num > 251) gen = 3;
                    if (num > 386) gen = 4;
                    if (num > 493) gen = 5;
                    if (num > 649) gen = 6;
                    if (num > 721) gen = 7;
                    if (num > 809) gen = 8;
                    if (num > 905) gen = 9;

                    return {
                        num: num,
                        name: cleanName,
                        gen: gen,
                        base: false,
                        male: false,
                        female: false,
                        shiny: false,
                        regional: false,
                        form: false
                    };
                });

                // Sort purely by ID to guarantee perfect database organization
                fetchedPokemon.sort((a, b) => a.num - b.num);

                document.getElementById('api-progress-bar').style.width = '60%';
                document.getElementById('api-progress-percent').innerText = '60%';
                document.getElementById('api-progress-label').innerText = "Merging local user changes...";

                // Map existing configurations to avoid deleting progress
                const stateMap = {};
                activePokemonList.forEach(p => {
                    stateMap[p.num] = p;
                });

                activePokemonList = fetchedPokemon.map(p => {
                    if (stateMap[p.num]) {
                        // Merge achievements preserving keys
                        return {
                            ...p,
                            base: stateMap[p.num].base || false,
                            male: stateMap[p.num].male || false,
                            female: stateMap[p.num].female || false,
                            shiny: stateMap[p.num].shiny || false,
                            regional: stateMap[p.num].regional || false,
                            form: stateMap[p.num].form || false
                        };
                    }
                    return p;
                });

                document.getElementById('api-progress-percent').innerText = '100%';
                document.getElementById('api-progress-label').innerText = "Sync Complete!";

        function applyFiltersAndRender() {
            const query = document.getElementById('dex-search').value.toLowerCase().trim();
            const genFilter = document.getElementById('dex-gen-filter').value;
            const stateFilter = document.getElementById('dex-state-filter').value;

            filteredPokemon = activePokemonList.filter(p => {
                // Search match
                const matchSearch = p.name.toLowerCase().includes(query) || p.num.toString().includes(query);
                
                // Gen filter match
                const matchGen = (genFilter === 'all') || (p.gen.toString() === genFilter);
                
                // State filters
                let matchState = true;
                if (stateFilter === 'caught') {
                    matchState = (p.base || p.male || p.female || p.shiny || p.regional || p.form);
                } else if (stateFilter === 'missing') {
                    matchState = !(p.base || p.male || p.female || p.shiny || p.regional || p.form);
                } else if (stateFilter === 'shiny') {
                    matchState = p.shiny;
                }

                return matchSearch && matchGen && matchState;
            });

            // Boundary adjustments
            const maxPages = Math.ceil(filteredPokemon.length / itemsPerPage) || 1;
            if (currentPage > maxPages) currentPage = maxPages;

            renderPage();
        }

        function toggleCardStat(pNum, key) {

        function importSyncString() {
            const input = document.getElementById('sync-import-area').value.trim();
            if (!input) {
                triggerToast("Paste a valid sync code block first!", 'error');
                return;
            }

            try {
                const decodedRaw = decodeURIComponent(escape(atob(input)));
                const parsed = JSON.parse(decodedRaw);
                
                if (parsed.missions || parsed.pokemon) {
                    if (parsed.missions) {
                        missionsState = parsed.missions;
                        // Map checklist inputs
                        Object.keys(missionsState).forEach(id => {
                            const checkbox = document.getElementById(id);
                            if (checkbox) checkbox.checked = missionsState[id];
                        });

                    if (parsed.pokemon && Array.isArray(parsed.pokemon)) {
                        activePokemonList = parsed.pokemon;
                    }
                    
                    saveLocalState();
                    applyFiltersAndRender();
                    document.getElementById('sync-import-area').value = "";
                    triggerToast("Sync data loaded successfully!", 'success');
                } else {
                    throw new Error("Invalid structures");
                }
